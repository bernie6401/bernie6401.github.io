    from PIL import Image

img = Image.open("./NISRA-Enlightened-2023/Misc/Final-Pixel.png")

	for col in range(8):	# 8 columns

		if r == 255 and g == 255 and b == 255:

		else:

	if r == 255 and g == 255 and b == 255:
		break

flag = ""
print(pixels)
for r, g, b in pixels:
	flag += r + g + b
print(flag)
print(b64decode(flag.encode()))
