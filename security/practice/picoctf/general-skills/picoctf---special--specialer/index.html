    Exploit #

    • 代替cat的方法：

      # Method 1
$ while read line; do
while> echo $line;
while> done <{filename}}

# Method 2
$ exec 3<{filename}} # Assign file descriptor 3 for reading
$ while read -u 3 line; do
while> echo $line
while> done

# Method 3
$ echo "$(<{filename})"

    • 代替ls的方法

      # Method 1
$ echo *

# Method 2
$ echo */*

# Method 3
$ echo * .*

# Method 4
$ dir

# Method 5
$ printf '%s\n' *

# Method 6
$ grep -l '.*' ./*

# Method 7
$ find .

# Method 8
$ stat -c '%s %A %n' *

# Method 9
$ lsattr ./*

# Method 10
$ vim .
